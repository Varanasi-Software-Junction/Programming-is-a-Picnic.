Lesson 1: Installing Angular and Setting Up Basic Routing

Objective

To install Angular CLI, create a new Angular project, and set up routing to navigate between multiple pages.

Step 1: Install Angular CLI

  1. Install Node.js (if not already installed). Download from Node.js.
  2. Install Angular CLI by running: 
                
npm install -g @angular/cli
  3. Verify the installation with: 
                
ng version

Step 2: Create a New Angular Project

  1. Initialize a new project: 
                
ng new my-angular-app
    Choose Yes for routing and CSS for styling.
  2. Navigate into the project directory: 
                
cd my-angular-app
  3. Start the development server: 
                
ng serve
    Visit http://localhost:4200/ in your browser to see the app.

Step 3: Set Up Basic Routing

  1. Create two components: 
                
ng generate component home
ng generate component about
  2. Open app-routing.module.ts and configure routes: 
                
// app-routing.module.ts
import { NgModule } from '@angular/core';
import { RouterModule, Routes } from '@angular/router';
import { HomeComponent } from './home/home.component';
import { AboutComponent } from './about/about.component';

const routes: Routes = [
    { path: '', component: HomeComponent },
    { path: 'about', component: AboutComponent }
];

@NgModule({
    imports: [RouterModule.forRoot(routes)],
    exports: [RouterModule]
})
export class AppRoutingModule { }
  3. Update app.component.html with navigation links: 
                
<nav>
    <a routerLink="/">Home</a> | 
    <a routerLink="/about">About</a>
</nav>
<router-outlet></router-outlet>

Example Files for Components

home.component.html

    
<h2>Welcome to the Home Page</h2>
<p>This is the main landing page.</p>

home.component.ts

    
import { Component } from '@angular/core';

@Component({
    selector: 'app-home',
    templateUrl: './home.component.html',
    styleUrls: ['./home.component.css']
})
export class HomeComponent {}

about.component.html

    
<h2>About Us</h2>
<p>Learn more about us on this page.</p>

about.component.ts

    
import { Component } from '@angular/core';

@Component({
    selector: 'app-about',
    templateUrl: './about.component.html',
    styleUrls: ['./about.component.css']
})
export class AboutComponent {}
