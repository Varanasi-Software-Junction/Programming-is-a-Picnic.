Understanding If-Else, Ternary Operator, and Switch in C

If-Else Statements

The if-else statement in C allows for decision-making based on conditions. The program can execute different code blocks depending on whether the condition is true or false.

Unsolved Questions:

1. Write a program to check if a number is even or odd using if-else.
#include 
int main() {
    int num;
    printf("Enter a number: ");
    scanf("%d", &num);

    if (num % 2 == 0) {
        printf("Even number\n");
    } else {
        printf("Odd number\n");
    }
    return 0;
}
2. Write an if-else program to check if a number is positive, negative, or zero.
#include 
int main() {
    int num;
    printf("Enter a number: ");
    scanf("%d", &num);

    if (num > 0) {
        printf("Positive\n");
    } else if (num < 0) {
        printf("Negative\n");
    } else {
        printf("Zero\n");
    }
    return 0;
}
3. Use if-else to check if a character entered by the user is a vowel or a consonant.
#include 
int main() {
    char ch;
    printf("Enter a character: ");
    scanf(" %c", &ch);

    if (ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u' ||
        ch == 'A' || ch == 'E' || ch == 'I' || ch == 'O' || ch == 'U') {
        printf("Vowel\n");
    } else {
        printf("Consonant\n");
    }
    return 0;
}
4. Write an if-else program to check if a year is a leap year.
#include 
int main() {
    int year;
    printf("Enter a year: ");
    scanf("%d", &year);

    if ((year % 4 == 0 && year % 100 != 0) || year % 400 == 0) {
        printf("Leap year\n");
    } else {
        printf("Not a leap year\n");
    }
    return 0;
}
5. Write a program using if-else to find the largest of three numbers.
#include 
int main() {
    int a, b, c;
    printf("Enter three numbers: ");
    scanf("%d %d %d", &a, &b, &c);

    if (a >= b && a >= c) {
        printf("%d is the largest\n", a);
    } else if (b >= a && b >= c) {
        printf("%d is the largest\n", b);
    } else {
        printf("%d is the largest\n", c);
    }
    return 0;
}

Ternary Operator

The ternary operator (?:) is a shorthand for the if-else statement. It evaluates a condition and returns one of two values depending on whether the condition is true or false.

Unsolved Questions:

1. Write a program to find the largest of two numbers using the ternary operator.
#include 
int main() {
    int a = 10, b = 20;
    int largest = (a > b) ? a : b;
    printf("Largest number is: %d\n", largest);
    return 0;
}
2. Write a program to check if a number is even or odd using the ternary operator.
#include 
int main() {
    int num;
    printf("Enter a number: ");
    scanf("%d", &num);
    printf("%d is %s\n", num, (num % 2 == 0) ? "Even" : "Odd");
    return 0;
}
3. Use the ternary operator to check if a number is positive, negative, or zero.
#include 
int main() {
    int num;
    printf("Enter a number: ");
    scanf("%d", &num);
    (num > 0) ? printf("Positive\n") : (num < 0) ? printf("Negative\n") : printf("Zero\n");
    return 0;
}
4. Use the ternary operator to assign the absolute value of a number to a variable.
#include 
int main() {
    int num = -5;
    int abs_value = (num < 0) ? -num : num;
    printf("Absolute value is: %d\n", abs_value);
    return 0;
}
5. Write a program using the ternary operator to check if a character is uppercase or lowercase.
#include 
int main() {
    char ch;
    printf("Enter a character: ");
    scanf(" %c", &ch);
    (ch >= 'A' && ch <= 'Z') ? printf("Uppercase\n") : printf("Lowercase\n");
    return 0;
}

Switch Statements

The switch statement in C is used to select one of many code blocks to execute, based on the value of an expression.

Unsolved Questions:

1. Write a program to display the day of the week based on a number (1 for Monday, 2 for Tuesday, etc.) using switch.
#include 
int main() {
    int day;
    printf("Enter a number (1-7): ");
    scanf("%d", &day);

    switch(day) {
        case 1: printf("Monday\n"); break;
        case 2: printf("Tuesday\n"); break;
        case 3: printf("Wednesday\n"); break;
        case 4: printf("Thursday\n"); break;
        case 5: printf("Friday\n"); break;
        case 6: printf("Saturday\n"); break;
        case 7: printf("Sunday\n"); break;
        default: printf("Invalid input\n");
    }
    return 0;
}
2. Write a switch statement to convert a character to lowercase or uppercase.
#include 
int main() {
    char ch;
    printf("Enter a character: ");
    scanf(" %c", &ch);

    switch(ch) {
        case 'a' ... 'z': printf("Lowercase\n"); break;
        case 'A' ... 'Z': printf("Uppercase\n"); break;
        default: printf("Not a letter\n");
    }
    return 0;
}
3. Write a program using switch to perform basic arithmetic operations (+, -, *, /).
#include 
int main() {
    char op;
    float num1, num2;

    printf("Enter an operator (+, -, *, /): ");
    scanf(" %c", &op);
    printf("Enter two operands: ");
    scanf("%f %f", &num1, &num2);

    switch(op) {
        case '+': printf("%.2f + %.2f = %.2f\n", num1, num2, num1 + num2); break;
        case '-': printf("%.2f - %.2f = %.2f\n", num1, num2, num1 - num2); break;
        case '*': printf("%.2f * %.2f = %.2f\n", num1, num2, num1 * num2); break;
        case '/': 
            if(num2 != 0)
                printf("%.2f / %.2f = %.2f\n", num1, num2, num1 / num2); 
            else
                printf("Division by zero error.\n");
            break;
        default: printf("Invalid operator\n");
    }
    return 0;
}
