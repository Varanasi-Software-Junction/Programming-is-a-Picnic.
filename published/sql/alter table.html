ALTER TABLE in Oracle 10G

1. Adding, Modifying, and Removing Columns

Adding a Column

ALTER TABLE employees ADD age NUMBER(3);

Modifying a Column

ALTER TABLE employees MODIFY age NUMBER(5);

Removing a Column

ALTER TABLE employees DROP COLUMN age;

2. Managing Constraints

Adding a Primary Key

ALTER TABLE employees ADD CONSTRAINT emp_pk PRIMARY KEY (emp_id);

Removing a Primary Key

ALTER TABLE employees DROP CONSTRAINT emp_pk;

Adding a Foreign Key

ALTER TABLE employees ADD CONSTRAINT emp_dept_fk FOREIGN KEY (dept_id) REFERENCES departments(dept_id);

Removing a Foreign Key

ALTER TABLE employees DROP CONSTRAINT emp_dept_fk;

Adding a Unique Constraint

ALTER TABLE employees ADD CONSTRAINT emp_email_unique UNIQUE (email);

Removing a Unique Constraint

ALTER TABLE employees DROP CONSTRAINT emp_email_unique;

3. Renaming Table and Columns

Renaming a Table

ALTER TABLE employees RENAME TO staff;

Renaming a Column

ALTER TABLE employees RENAME COLUMN salary TO base_salary;

4. Managing Default Values

Adding a Default Value

ALTER TABLE employees MODIFY salary NUMBER(10,2) DEFAULT 50000;

Removing a Default Value

ALTER TABLE employees MODIFY salary DEFAULT NULL;

5. Managing NOT NULL Constraints

Adding NOT NULL

ALTER TABLE employees MODIFY email VARCHAR2(100) NOT NULL;

Removing NOT NULL

ALTER TABLE employees MODIFY email NULL;

Summary of ALTER TABLE Operations
