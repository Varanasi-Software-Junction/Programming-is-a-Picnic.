Operator Overloading in C++

Operator overloading allows user-defined types to behave like built-in types by redefining the way operators work for those types.

List of Operators That Can Be Overloaded

  • Arithmetic Operators: +, -, *, /
  • Comparison Operators: ==, !=, <,>, <=,>=
  • Assignment Operators: =, +=, -=, *=, /=
  • Bitwise Operators: &, |, ^, ~, <<,>>
  • Increment/Decrement Operators: ++, --
  • Logical Operators: &&, ||
  • Subscript Operator: []
  • Function Call Operator: ()
  • Pointer Operators: ->, *

Examples of Overloading Operators

1. Overloading the + Operator

#include 
using namespace std;
class Complex {
public:
    int real, imag;
    Complex(int r, int i) : real(r), imag(i) {}
    Complex operator+(const Complex& obj) {
        return Complex(real + obj.real, imag + obj.imag);
    }
};
int main() {
    Complex c1(2, 3), c2(4, 5);
    Complex c3 = c1 + c2;
    cout << "Sum: " << c3.real << "+" << c3.imag << "i";
    return 0;
}

2. Overloading the < Operator

#include 
using namespace std;
class Number {
public:
    int value;
    Number(int v) : value(v) {}
    bool operator<(const Number& obj) {
        return value < obj.value;
    }
};
int main() {
    Number n1(10), n2(20);
    if (n1 < n2)
        cout << "n1 is less than n2";
    else
        cout << "n1 is not less than n2";
    return 0;
}

3. Overloading the <= Operator

#include 
using namespace std;
class Number {
public:
    int value;
    Number(int v) : value(v) {}
    bool operator<=(const Number& obj) {
        return value <= obj.value;
    }
};
int main() {
    Number n1(10), n2(10);
    if (n1 <= n2)
        cout << "n1 is less than or equal to n2";
    else
        cout << "n1 is greater than n2";
    return 0;
}
